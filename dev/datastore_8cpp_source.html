123  void DataStore::saveAgentInstance( const double time, const AgentId & id,
124  const map<string, const ScriptValue *> & vars )
125  {
126  if( vars.size() == 0 || !m_isConnected ){
127  return;
128  }
129 
130  string ns{ m_dbname + AGENT_COLL_PREFIX + "m" + to_string(id.m_mayor) };
131 
132  bson b;
133 
134  bson_init( &b );
135  if( bson_append_new_oid( &b, "_id" ) != BSON_OK ){
136  LOGE( "failed to create bson id" );
137  mongo_disconnect( &m_conn );
138  return;
139  }
140 
141  bson_append_int( &b, "minor", id.m_minor );
142  bson_append_double( &b, "time", time );
143 
144  for( const auto& i: vars ){
145  const ScriptValue & val = *(i.second);
146  switch( val.getType() ){
147  case ScriptValue::ValueType::NIL: // empty
148  break;
149  case ScriptValue::ValueType::BOOLEAN:
150  bson_append_bool( &b, i.first.c_str(), val.getBoolean() );
151  break;
152  case ScriptValue::ValueType::NUMBER:
153  bson_append_double( &b, i.first.c_str(), val.getNumber() );
154  break;
155  case ScriptValue::ValueType::STRING:
156  bson_append_string( &b, i.first.c_str(), val.getString().c_str() );
157  break;
158  }
159  }
160 
161  bson_finish( &b );
162 
163  if( mongo_insert( &m_conn , ns.c_str() , &b, NULL ) != MONGO_OK ){
164  LOGE( "failed to insert in mongo" );
165  }
166 
167  bson_destroy( &b );
168  }

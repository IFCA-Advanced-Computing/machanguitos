51  lua_getfield( L, -1, "init"); // 2
52  if( lua_isfunction( L, -1 ) ){
53  lua_getfield( L, LUA_GLOBALSINDEX, SCRIPT_AGENT_NAME ); // 3

55  lua_setglobal( L, SCRIPT_GLOBAL_AGENT_OBJ ); // 3
56  auto ret = lua_pcall( L, 1, 0, 0 ); // 1
57  checkLuaReturn( L, ret );
58  lua_pop( L, 1 ); // 0
59  }else{
60  lua_pop( L, 2 ); // 0
61  }
62  }

89  int AgentInstance::pushData( lua_State * L, const string & key ) const{
90  const auto i = m_vals.find(key);
91  if( i != m_vals.end() ){
92  switch( i->second.getType() ){
93  case ScriptValue::ValueType::NIL:
94  lua_pushnil( L );
95  break;
96 
97  case ScriptValue::ValueType::BOOLEAN:
98  lua_pushboolean( L, i->second.getBoolean() );
99  break;
100 
101  case ScriptValue::ValueType::NUMBER:
102  lua_pushnumber( L, i->second.getNumber() );
103  break;
104 
105  case ScriptValue::ValueType::STRING:
106  lua_pushstring( L, i->second.getString().c_str() );
107  break;
108 
109  default:
110  lua_pushnil( L );
111  break;
112  }
113  }else{
114  lua_pushnil( L );
115  }
116  return 1;

120  void AgentInstance::newData( lua_State * L, const string & key ){
121  auto ltype = lua_type( L, -1 );
122  switch( ltype ){
123  case LUA_TNIL:
124  m_vals[key] = ScriptValue();
125  break;
126  case LUA_TNUMBER:
127  m_vals[key] = ScriptValue(lua_tonumber( L, -1 ));
128  break;
129  case LUA_TBOOLEAN:
130  m_vals[key] = ScriptValue(lua_toboolean( L, -1 )==1);
131  break;
132  case LUA_TSTRING:
133  m_vals[key] = ScriptValue(lua_tostring( L, -1));
134  break;
135  default:
136  LOGW( "Lua type not implemented '", lua_typename( L, ltype ),
137  "' on key '", key, "'" );
138  }

46  bool MPIClientServer::doTags( int tag, int src, int32_t val ){
47  switch( tag ){
48  case MpiTagCS::CREATECLASS:
49  runCreateClass( src );
50  break;
51 
52  case MpiTagCS::CREATEAGENTS:
53  runCreateAgents( src, val );
54  break;
55 
56  case MpiTagCS::RUNAGENTS:
57  runAgents( src );
58  break;
59 
60  case MpiTagCS::SETSTARTTIME:
61  runSetStartTime( src );
62  break;
63 
64  case MpiTagCS::SETRANDOMSEED:
65  Util::LOGD( "Set client ", m_rank, " seed ", val );
66  m_local->setRandomSeed( val );
67  break;
68 
69  case MpiTagCS::CREATERASTERCLIENT:
70  runCreateRasterClient( src, val );
71  break;
72 
73  case MpiTagCS::LOADRASTERCLIENT:
74  runLoadRasterClient( src );
75  break;
76 
77  default:
78  return false;
79  }
80 
81  return true;
82  }

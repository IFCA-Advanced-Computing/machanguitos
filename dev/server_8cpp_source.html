117  void Server::createAgents(){
118  auto nClients = m_clients.size();
119  if( nClients > 0 ){
120  for( const auto & p: m_numAgents ){
121  // sort, first the clients with less agents
122  sort( begin(m_clients), end(m_clients),
123  []( const unique_ptr<Client> & a,
124  const unique_ptr<Client> & b)
125  {
126  return a->numAgents() < b->numAgents();
127  } );
128 
129  LOGI( "Spawning: ", p.second, " of '", p.first, "'" );
130  decltype(nClients) nPerClient = p.second / nClients;
131  decltype(nClients) rem = p.second % nClients;
132 
133  // put more agents in the first clients
134  decltype(nClients) i;
135  for( i = 0 ; i < rem ; ++i ){
136  auto & c = m_clients[i];
137  if( c->createClass( p.first ) ){
138  c->createAgents( p.first, nPerClient+1 );
139  }else{
140  LOGW( "Class '", p.first, "' can't be created" );
141  }
142  }
143 
144  // then put the rest
145  for( i = rem ; i < m_clients.size() ; ++i ){
146  auto & c = m_clients[i];
147  if( c->createClass( p.first ) ){
148  c->createAgents( p.first, nPerClient );
149  }else{
150  LOGW( "Class '", p.first, "' can't be created" );
151  }
152  }
153  }
154  }else{
155  LOGW( "No clients" );
156  }
157  }

199  void Server::run(){
200  auto iters = getConfigInt( "iters", 10 );
201  auto startt = getConfigNumber( "starttime", 0 );
202  auto endt = getConfigNumber( "endtime", 10 );
203 
204  double delta = iters > 0 ? (endt - startt) / static_cast<double>(iters) : 0;
205 
206  LOGI( "SERVER: Start Simulation\n" );
207  for( int i = 0 ; i < iters ; i++ ){
208  LOGI( "SERVER: iteration ", i+1 );
209  for( auto && c: m_clients ){
210  c->runAgents( delta );
211  }
212 
213  clientsBarrier();
214  }
215 
216  remoteEnd();
217 
218  LOGI( "SERVER: End Simulation\n" );
219  }

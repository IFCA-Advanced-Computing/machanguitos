135  LOGE( "There is no config file '", filename, "'" );

207  void Server::createAgents(){
208  auto nClients = m_clients.size();
209  if( nClients > 0 ){
210  for( const auto & p: m_numAgents ){
211  // sort, first the clients with less agents
212  sort( begin(m_clients), end(m_clients),
213  []( const unique_ptr<Client> & a,
214  const unique_ptr<Client> & b)
215  {
216  return a->numAgents() < b->numAgents();
217  } );
218 
219  LOGI( "Spawning: ", p.second, " of '", p.first, "'" );
220  decltype(nClients) nPerClient = p.second / nClients;
221  decltype(nClients) rem = p.second % nClients;
222 
223  // put more agents in the first clients
224  decltype(nClients) i;
225  for( i = 0 ; i < rem ; ++i ){
226  auto & c = m_clients[i];
227  if( c->createClass( p.first ) ){
228  c->createAgents( p.first, nPerClient+1 );
229  }else{
230  LOGW( "Class '", p.first, "' can't be created" );
231  }
232  }
233 
234  // then put the rest
235  for( i = rem ; i < m_clients.size() ; ++i ){
236  auto & c = m_clients[i];
237  if( c->createClass( p.first ) ){
238  c->createAgents( p.first, nPerClient );
239  }else{
240  LOGW( "Class '", p.first, "' can't be created" );
241  }
242  }
243  }
244  }else{
245  LOGW( "No clients" );
246  }
247  }

289  void Server::run(){
290  auto iters = getConfigInt( "iters", 10 );
291  auto startt = getConfigNumber( "starttime", 0 );
292  auto endt = getConfigNumber( "endtime", 10 );
293  auto ds = Engine::DataServer::instance();
294 
295  double delta = iters > 0 ? (endt - startt) / static_cast<double>(iters) : 0;
296 
297  LOGI( "SERVER: Start Simulation\n" );
298  for( auto i = 0 ; i < iters ; ++i ){
299  startIterationScript( i+1 );
300  for( auto && c: m_clients ){
301  c->runAgents( delta );
302  }
303 
304  clientsBarrier();
305  ds->updateLayers( delta );
306  endIterationScript( i+1 );
307  }
308 
309  remoteEnd();
310 
311  endSimulationScript();
312  }

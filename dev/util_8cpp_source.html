89  unique_ptr<string> getGDALDriverName( const string & filename ){
90  using Value = string::value_type;
91  auto lower = [](const Value c){ return (Value)tolower(c); };
92 
93  // get the lowercase extension of filename
94  auto fileext = path( filename ).extension().generic_string();
95  string normext;
96  normext.resize( fileext.size() );
97  transform( begin(fileext), end(fileext), begin(normext), lower );
98 
99  // search the extension in the driver list
100  for( const auto di: s_driverNames ){
101  auto el = find( begin(di.driverExts), end(di.driverExts), normext);
102  if( el != end(di.driverExts) ){
103  return unique_ptr<string>(new string(di.driverName));
104  }
105  }
106

41  void MPIWorker::run(){
42  int32_t val{0};
43  MPI_Status status;
44  m_running = true;
45 
46  while( m_running ){
47  MPI_Recv( &val, 1, MPI_INT, MPI_ANY_SOURCE, MPI_ANY_TAG, MPI_COMM_WORLD, &status );
48  if( status.MPI_ERROR != MPI_SUCCESS ){
49  LOGE( "Received on ", m_rank );
50  Engine::abort();
51  }
52 
53  bool is_checked = doCommonTags( status.MPI_TAG, status.MPI_SOURCE, val );
54  if( not is_checked ){
55  is_checked = doTags( status.MPI_TAG, status.MPI_SOURCE, val );
56  }
57 
58  if( not is_checked ){
59  LOGE( "Not-implemented message[", status.MPI_TAG, "] on ", m_rank,
60  " from ", status.MPI_SOURCE );
61  }
62  }
63 
64  }

67  bool MPIWorker::doCommonTags( int tag, int source, int32_t val ){
68  switch( tag ){
69  case MpiTag::SETDATASTORE:
70  runSetDataStore( source, val );
71  break;
72 
73  case MpiTag::SETDATAPATH:
74  runSetDataPath( source );
75  break;
76 
77  case MpiTag::SETLOGLEVEL:
78  setLogLevel( val );
79  break;
80 
81  case MpiTag::END:
82  m_running = false;
83  break;
84 
85  default:
86  return false;
87  }
88 
89  return true;

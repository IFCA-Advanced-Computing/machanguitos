50  void MPIWorker::run(){
51  int32_t val{0};
52  MPI_Status status;
53  bool running = true;
54 
55  while( running ){
56  MPI_Recv( &val, 1, MPI_INT, 0, MPI_ANY_TAG, MPI_COMM_WORLD, &status );
57  if( status.MPI_ERROR != MPI_SUCCESS ){
58  LOGE( "Received on ", m_rank );
59  Engine::abort();
60  }
61 
62  switch( status.MPI_TAG ){
63  case MpiTagCS::CREATECLASS:
64  runCreateClass();
65  break;
66 
67  case MpiTagCS::CREATEAGENTS:
68  runCreateAgents( val );
69  break;
70 
71  case MpiTagCS::RUNAGENTS:
72  runAgents();
73  break;
74 
75  case MpiTagCS::SETSTARTTIME:
76  runSetStartTime();
77  break;
78 
79  case MpiTagCS::SETDATASTORE:
80  runSetDataStore( val );
81  break;
82 
83  case MpiTagCS::CREATERASTERCLIENT:
84  runCreateRasterClient( val );
85  break;
86 
87  case MpiTag::SETDATAPATH:
88  runSetDataPath();
89  break;
90 
91  case MpiTag::SETLOGLEVEL:
92  setLogLevel( val );
93  break;
94 
95  case MpiTag::END:
96  running = false;
97  break;
98 
99  default:
100  LOGE( "Not-implemented message[", status.MPI_TAG, "] on ", m_rank );
101  }
102  }
103  }

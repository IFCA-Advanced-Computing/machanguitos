93  int config_setvars( lua_State *L ){
94  auto server = Engine::Server::instance();
95  luaL_checktype( L, 1, LUA_TTABLE );
96  // push first key
97  lua_pushnil( L );
98  while( lua_next( L, 1 ) != 0 ) {
99  string key( luaL_checkstring( L, -2 ) );
100  auto ltype = lua_type( L, -1 );
101  switch( ltype ){
102  case LUA_TNIL:
103  server->insertConfig( key, ScriptValue() );
104  break;
105  case LUA_TNUMBER:
106  server->insertConfig( key, ScriptValue(lua_tonumber( L, -1 )) );
107  break;
108  case LUA_TBOOLEAN:
109  server->insertConfig( key, ScriptValue(lua_toboolean( L, -1)==1) );
110  break;
111  case LUA_TSTRING:
112  server->insertConfig( key, ScriptValue(lua_tostring( L, -1)) );
113  break;
114  default:
115  luaL_warn( L, "type not implemented '%s' on key '%s'",
116  lua_typename( L, ltype ), key.c_str() );
117  }
118 
119  // removes 'value'; keeps 'key' for next iteration
120  lua_pop( L, 1 );
121  }
122  return 0;
123  }

00033         auto server = Engine::Server::instance();
00034         luaL_checktype( L, 1, LUA_TTABLE );
00035         // push first key
00036         lua_pushnil( L );
00037         while( lua_next( L, 1 ) != 0 ) {
00038             std::string key( luaL_checkstring( L, -2 ) );
00039             auto ltype = lua_type( L, -1 );
00040             switch( ltype ){
00041             case LUA_TNIL:
00042                 server->insertConfig( key, ScriptValue() );
00043                 break;
00044             case LUA_TNUMBER:
00045                 server->insertConfig( key, ScriptValue(lua_tonumber( L, -1 )) );
00046                 break;
00047             case LUA_TBOOLEAN:
00048                 server->insertConfig( key, ScriptValue(lua_toboolean( L, -1)==1) );
00049                 break;
00050             case LUA_TSTRING:
00051                 server->insertConfig( key, ScriptValue(lua_tostring( L, -1)) );
00052                 break;
00053             default:
00054                 cerr << "WARNING: type not implemented '" << lua_typename( L, ltype )
00055                      << "' on key '" << key << "'\n";
00056             }
00057 
00058             // removes 'value'; keeps 'key' for next iteration
00059             lua_pop( L, 1 );
00060         }
00061         return 0;
00062     }

00082     int config_setvars( lua_State *L ){
00083         auto server = Engine::Server::instance();
00084         luaL_checktype( L, 1, LUA_TTABLE );
00085         // push first key
00086         lua_pushnil( L );
00087         while( lua_next( L, 1 ) != 0 ) {
00088             std::string key( luaL_checkstring( L, -2 ) );
00089             auto ltype = lua_type( L, -1 );
00090             switch( ltype ){
00091             case LUA_TNIL:
00092                 server->insertConfig( key, ScriptValue() );
00093                 break;
00094             case LUA_TNUMBER:
00095                 server->insertConfig( key, ScriptValue(lua_tonumber( L, -1 )) );
00096                 break;
00097             case LUA_TBOOLEAN:
00098                 server->insertConfig( key, ScriptValue(lua_toboolean( L, -1)==1) );
00099                 break;
00100             case LUA_TSTRING:
00101                 server->insertConfig( key, ScriptValue(lua_tostring( L, -1)) );
00102                 break;
00103             default:
00104                 luaL_warn( L, "type not implemented '%s' on key '%s'",
00105                            lua_typename( L, ltype ), key.c_str() );
00106             }
00107 
00108             // removes 'value'; keeps 'key' for next iteration
00109             lua_pop( L, 1 );
00110         }
00111         return 0;
00112     }
